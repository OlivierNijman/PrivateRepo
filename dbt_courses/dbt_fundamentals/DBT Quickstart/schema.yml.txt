  - name: customers
    description: One record per customer
    columns:
      - name: customer_id
        description: Primary key
        tests:
          - unique
          - not_null
      - name: first_name
        description: First name of customer
      - name: last_name
        description: Last name of customer
      - name: first_order_date
        description: First date customer placed an order, NULL if no orders from customer
      - name: most_recent_order_date 
        description: Most recent customer order date, NULL if no orders from customer
      - name: number_of_orders
        description: Number of orders, 0 if no orders
          
  - name: stg_customers
    description: This model gives clean version of customer data
    columns:
      - name: customer_id
        description: Primary key
        tests:
          - unique
          - not_null
      - name: first_name
        description: First name of customer
      - name: last_name
        description: Last name of customer
  
  - name: stg_orders
    description: This model gives clean version of order data
    columns:
      - name: order_id
        description: Primary key
        tests:
          - unique
          - not_null
      - name: status
        description: Status of order, tested for accepted values
        tests:
          - accepted_values:
              values: ['placed', 'shipped', 'completed', 'return_pending', 'returned']
      - name: customer_id
        description: Customer id to link with customers
        tests:
          - not_null
          - relationships:
              to: ref('stg_customers')
              field: customer_id
      - name: order_date
        description: Date the order was placed
